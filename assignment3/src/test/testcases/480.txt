gcdRecursion: function integer(p: integer, q: integer){
            if (p > q)
                return gcdRecursion(p - q, q);
            else
                if (p < q)
                    return gcdRecursion(p, q - p);
                else
                    return p;
            }
        gcdIteration: function integer(p: integer, q: integer){
            while (p != q) {
                if (p > q)
                    p = p - q;
                else
                    q = q - p;
                }
            return p;
            }
